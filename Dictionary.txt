2.your_expenses = {
    "Hotel": 1200,
    "Food": 800,
    "Transportation": 500,
    "Attractions": 300,
    "Miscellaneous": 200
}

partner_expenses = {
    "Hotel": 1000,
    "Food": 900,
    "Transportation": 600,
    "Attractions": 400,
    "Miscellaneous": 150
}

your_total = sum(your_expenses.values())
partner_total = sum(partner_expenses.values())

print(f"Your total expenses: ₹{your_total}")
print(f"Partner's total expenses: ₹{partner_total}")

if your_total > partner_total:
    print("You spent more money overall.")
elif partner_total > your_total:
    print("Your partner spent more money overall.")
else:
    print("You both spent the same amount overall.")

max_diff_category = max(your_expenses, key=lambda k: abs(your_expenses[k] - partner_expenses[k]))
diff = abs(your_expenses[max_diff_category] - partner_expenses[max_diff_category])
print(f"Largest difference in '{max_diff_category}': ₹{diff}")
